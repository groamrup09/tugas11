SESI 11: Redirecting Constructors

Review Sesi 10 Factory Constructors

  • Factory constructor:factory Class.name()
  • Bisa return instance yang sudah ada atau subclass
  • Use cases: singleton, caching, object pools
  • 🐞 Contoh cepat:
factory Shape.circle() => Circle();

Pengantar Redirecting Constructors

  • Apa itu? Constructor yang "mengalihkan" ke constructor lain di kelas yang sama
  • Tujuan: Mengurangi duplikasi kode (DRY principle)
  • Sintaks: this(...) di initializer list
  • Tidak punya body (tidak ada { } )

Solusi dengan Redirecting Constructor

// Nama: Azriel Miracle Yuandre
// Nim: 251410024
// Kelas: SI2A
class Rectangle {
final double width;
final double height;
// Main constructor
Rectangle(this.width, this.height) {
print('Rectangle $width x $height created');
}
// Redirecting constructor
Rectangle.square(double side) : this(side, side);
Rectangle.withDefault() : this(100, 50);
Rectangle.fromWidth(double width) : this(width, width * 0.618); // Golden ratio
@override
String toString() {
return 'Rectangle(width: ${width.toStringAsFixed(2)}, height: ${height.toStringAsFixed(2)})';
}
}
void main(){
// Penggunaan
var square = Rectangle.square(10);      
// 10x10
var defaultRect = Rectangle.withDefault(); // 100x50
var goldenRect = Rectangle.fromWidth(100); // 100x61.8
print('Square: $square');
print('Default Rectangle: $defaultRect');
print('Golden Ratio Rectangle: $goldenRect');
}

class Person {
  final String name;
  final int age;
  final String? email;

  // Main constructor dengan named parameters
  Person({required this.name, required this.age, this.email});

  // Redirecting constructors dengan default values
  Person.anonymous() : this(name: 'Anonymous', age: 0);

  Person.fromName(String name) : this(name: name, age: 0);

  Person.adult(String name) : this(name: name, age: 18);

  Person.child(String name, int years)
      : this(name: name, age: years, email: null);

  @override
  String toString() {
    return 'Name: $name, Age: $age, Email: ${email ?? "N/A"}';
  }
}

void main() {
  // Main constructor
  Person p1 = Person(
    name: "Azriel",
    age: 19,
    email: "azriel@gmail.com",
  );

  // Anonymous constructor
  Person p2 = Person.anonymous();

  // FromName constructor
  Person p3 = Person.fromName("Budi");

  // Adult constructor
  Person p4 = Person.adult("Siti");

  // Child constructor
  Person p5 = Person.child("Andi", 10);

  print("Data Person:");
  print(p1);
  print(p2);
  print(p3);
  print(p4);
  print(p5);
}

class BankAccount {
  final String accountNumber;
  final String ownerName;
  final double balance;

  BankAccount._({
    required this.accountNumber,
    required this.ownerName,
    required this.balance,
  });

  factory BankAccount.create({
    required String accountNumber,
    required String ownerName,
    double initialBalance = 0,
  }) {
    if (accountNumber.isEmpty) {
      throw ArgumentError('Nomor akun tidak boleh kosong');
    }
    if (ownerName.isEmpty) {
      throw ArgumentError('Nama pemilik tidak boleh kosong');
    }
    if (initialBalance < 0) {
      throw ArgumentError('Saldo awal tidak boleh negatif');
    }

    return BankAccount._(
      accountNumber: accountNumber,
      ownerName: ownerName,
      balance: initialBalance,
    );
  }

  factory BankAccount.fromJson(Map json) {
    try {
      return BankAccount.create(
        accountNumber: json['accountNumber']?.toString() ?? '',
        ownerName: json['ownerName']?.toString() ?? '',
        initialBalance: (json['balance'] as num?)?.toDouble() ?? 0,
      );
    } on ArgumentError catch (e) {
      throw FormatException('Data JSON tidak valid untuk BankAccount: ${e.message}');
    } catch (e) {
      throw FormatException('Kesalahan saat mem-parsing JSON untuk BankAccount: $e');
    }
  }

  @override
  String toString() {
    return 'Nomor Akun: $accountNumber, Pemilik: $ownerName, Saldo: \$${balance.toStringAsFixed(2)}';
  }
}

void main() {
  var account = BankAccount.create(
    accountNumber: 'ABC123',
    ownerName: 'Budi Santoso',
    initialBalance: 500000,
  );
  print(account);
}

🚀 Redirecting Constructor · Ringkasan & Best Practices

📌 Ringkasan:

  • Redirecting constructor: : this(...) tanpa body
  • Memanggil constructor lain di class yang sama
  • ✅ Mengurangi duplikasi kode (DRY principle)
  • ❌ Tidak bisa berisi logic kompleks (hanya delegasi)
  • Final fields tetap bisa diinisialisasi dengan rapi
  • Ideal untuk: default parameters, convenience constructors

✨ Best Practices:

  • 📐 Keep chains short — Hindari rantai redirecting yang terlalu panjang
  • 📛 Nama konstruktor bermakna: .fromJson(), .empty(), .withDefault()
  • 📄 Dokumentasi jelas: Tambahkan komentar pada tiap konstruktor pengarah
  • ⚡ Redirecting vs Factory: Redirecting = delegasi murni • Factory = logic kompleks / caching
  • 🧩 Gunakan untuk parameter default dan variasi objek tanpa duplikasi inisialisasi
  • 🔁 Jangan滥用 logic — Jika perlu validasi rumit, lebih baik gunakan factory + private constructor

class ApiResponse {
  final bool success;
  final T? data;
  final String? error;
  final int statusCode;
  final DateTime timestamp;

  // Main constructor
  ApiResponse({
    required this.success,
    this.data,
    this.error,
    required this.statusCode,
    required this.timestamp,
  });

  // Success response
  ApiResponse.success(T data)
      : this(
          success: true,
          data: data,
          statusCode: 200,
          timestamp: DateTime.now(),
        );

  // Error response
  ApiResponse.error(String error, int statusCode)
      : this(
          success: false,
          error: error,
          statusCode: statusCode,
          timestamp: DateTime.now(),
        );

  // Not found
  ApiResponse.notFound()
      : this(
          success: false,
          error: 'Resource not found',
          statusCode: 404,
          timestamp: DateTime.now(),
        );

  // Unauthorized
  ApiResponse.unauthorized()
      : this(
          success: false,
          error: 'Unauthorized access',
          statusCode: 401,
          timestamp: DateTime.now(),
        );

  // From JSON
  factory ApiResponse.fromJson(
    Map json,
    T Function(dynamic) fromJsonT,
  ) {
    return ApiResponse(
      success: json['success'] ?? false,
      data: json['data'] != null ? fromJsonT(json['data']) : null,
      error: json['error'],
      statusCode: json['statusCode'] ?? 500,
      timestamp: DateTime.parse(
        json['timestamp'] ?? DateTime.now().toIso8601String(),
      ),
    );
  }

  @override
  String toString() {
    return 'ApiResponse(success: $success, data: $data, error: $error, statusCode: $statusCode, timestamp: $timestamp)';
  }
}

void main() {
  var success = ApiResponse.success('Data loaded');
  var error = ApiResponse.error('Server error', 500);
  var notFound = ApiResponse.notFound();
  var unauthorized = ApiResponse.unauthorized();

  print(success);
  print(error);
  print(notFound);
  print(unauthorized);
}

class User {
  final String id;
  final String username;
  final String email;
  final DateTime createdAt;
  final bool isActive;

  User({
    required this.id,
    required this.username,
    required this.email,
    required this.createdAt,
    required this.isActive,
  });

  User.guest()
      : this(
          id: 'guest_${DateTime.now().millisecondsSinceEpoch}',
          username: 'Guest',
          email: 'guest@example.com',
          createdAt: DateTime.now(),
          isActive: false,
        );

  User.fromUsername(String username)
      : this(
          id: 'user_${username.toLowerCase()}',
          username: username,
          email: '${username.toLowerCase()}@example.com',
          createdAt: DateTime.now(),
          isActive: true,
        );

  factory User.create(String email, String password) {
    if (!email.contains('@')) {
      throw ArgumentError('Invalid email');
    }

    if (password.length < 8) {
      throw ArgumentError('Password too short');
    }

    return User(
      id: 'user_${DateTime.now().millisecondsSinceEpoch}',
      username: email.split('@').first,
      email: email,
      createdAt: DateTime.now(),
      isActive: true,
    );
  }

  @override
  String toString() {
    return 'User(id: $id, username: $username, email: $email, createdAt: $createdAt, isActive: $isActive)';
  }
}

void main() {
  User guest = User.guest();
  User user1 = User.fromUsername('Azriel');
  User user2 = User.create('azriel@gmail.com', 'password123');

  print(guest);
  print(user1);
  print(user2);
}

class Circle {

  // Redirecting dari diameter
  Circle.fromDiameter(double diameter) : this(diameter / 2);

  // Redirecting dari keliling

  String toString() {

  }
}

void main() {

  var circle2 = Circle.fromDiameter(10);

  print(circle1);
  print(circle2);
  print(circle3);
}
